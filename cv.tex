% \pill{TEXT}
%   draws text within an outline
\NewDocumentCommand{\pill}{ m }{%
  \begingroup%
  \size{7}%
  \tikz[baseline=-0.5ex]{
    \node[
      text=page,
      draw=section,
      fill=section!80,
      minimum height=1.7em,
      minimum width=2em,
      inner xsep=0.6em,
      inner ysep=0.6em,
      line width=0.15em,
      text height=0.8em,
      text depth=0.2em,
      rounded corners=1em,
    ] {#1};
  }%
  \endgroup%
}

% \header{ICON}[STYLE]{TEXT}
%   draws a bar spanning the column and a left-aligned icon and text
% \header{ICON}{TEXT}
%   default: STYLE = 'solid'
\NewDocumentCommand{\header}{ m O{solid} m }{%
  \begingroup%
  \size{14}%
  \vspace{1mm}%
  \par%
  \colorbox{section}{%
    \begin{minipage}[c][1em][b]{\columnwidth}%
      \begin{tabularx}{\columnwidth}{@{}m{2em}@{}X@{}}%
        \centering\color{section-icon}\icon[1.25]{#1}[#2] &%
        \color{page}#3%
      \end{tabularx}%
    \end{minipage}%
  }%
  \par%
  \endgroup%
}

% \role{ROLE}{PLACE}{CITY}{START}[END][SPLIT]
%   writes out a work/education entry information
%   the column length ratio is defined by SPLIT (first column takes 0.7 out of 1 by default)
% \role{ROLE}{PLACE}{CITY}{START}[END]
%   default: SPLIT = 0.7
% \role{ROLE}{PLACE}{CITY}{START}
%   default: END = '\tl{ongoing}'
\NewDocumentCommand{\role}{ m m m m O{\tl{ongoing}} O{0.7} }{%
  \par\vspace{1mm}%
  \begin{minipage}[c]{#6\columnwidth}%
    {\size{11}\bfseries #1} \\%
    {\color{section-compound}\icon[1.2]{at}} #2%
  \end{minipage}%
  \begin{minipage}[c]{\columnwidth-#6\columnwidth}%
    \raggedleft\size{8}%
    #3 \\%
    #4 - #5%
  \end{minipage}%
  \separate%
  \par%
}

  #3%
  \par%
  \endgroup%

    \vspace{4mm}

    \verb!\size{NUM}!

    Will change the size of the text in \textit{pt}, \textbf{10} is the default.

    \begingroup\color{gray}\verb!\size{7} some text!:\endgroup\ \begingroup\size{7} some text\endgroup

    You can delimit where the size is changed by using:

    \begingroup\bfseries\verb!\begingroup!\endgroup\ \verb!\size{12}! \begingroup\size{12} big text here \endgroup\begingroup\bfseries\verb!\endgroup!\endgroup

    (this is a \TeX\ feature and will work for other styles)

    \vspace{4mm}

    \verb!\whenpdf{YES}{NO}!

    Prints the content of \textit{YES} when compiling the PDF version and~\textit{NO}~when compiling the PNG version.
    The version is selected with the \textit{CVVER} environment variable. The provided compile script makes both.

    The clickable links use it to display as normal text in image form, since they're not clickable on the image anyway.

    You can use it for anything else, it allows putting whole paragraphs inside {\scriptsize(long type arguments)}.

    \begingroup\color{gray}\verb!\whenpdf{PDF}{PNG}!:\endgroup

    You are currently looking at the \textbf{\whenpdf{PDF}{PNG}} version of the document.

    \vspace{4mm}

    \verb!\tl{ID}[FORMAT]!

    Inserts translated text (the translations are set in the YAML file), optionally formatting it.

    The language is selected with the \textit{CVLANG} environment flag. The provided compile script can be set to compile all desired language variants (line 2).

    \begin{itemize}
      \item \textbf{normal:} Doesn't change anything in the text (don't specify the format for this)

      \item \textbf{lower:} prints the text as lowercase

    \verb!\icon[SIZE]{ICON}[STYLE]!

    Inserts a FontAwesome icon by name. Converts it to an image first, so that it is not seen by bots as text.

    The icon is scaled to the current text size at \textit{SIZE} = 1 (which is the default). Values above 1 grow it, and vice versa.

    The style parameter selects between available FontAwesome font variants.

    Inserts a link, the link is clickable and highlighted in the~\textit{PDF} version.

    \verb!\starbar[FRAC]!

    Draws a progress bar. When full, marks it with a star icon. The default value is full.

    The background color of the bar matches the background of~the~sidebar, it's best used as a highlighted skills section.

    (amount of stars / 5 apparently leads to negative connotations, a continuum does not)

    Refers to a work or education section. END is an optional end time, defaulting to a translated version of ``ongoing''.

    SPLIT is the fraction of space given to the left column, by~default 0.7 (this is to allow for very long texts with some manual tweaking).
